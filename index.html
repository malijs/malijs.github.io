# Full server in a few lines of code...

const Mali = require('mali')

async function sayHello (ctx) {
  ctx.res = { message: 'Hello '.concat(ctx.req.name) }
}

function main () {
  const app = new Mali('helloworld.proto')
  app.use({ sayHello })
  app.start('127.0.0.1:50051')
}
